## Тест-дизайн

```python
def test_compress_reduces_tokens():
    # Arrange
    large_context = "x " * 5000  # ~5000 tokens
    
    # Act
    compressed = compress_context(large_context, max_tokens=1000)
    
    # Assert
    assert len(compressed.split()) <= 1100  # with tolerance
    
def test_deduplication_removes_duplicates():
    # Arrange
    chunks = [
        {"content": "def auth():", "score": 0.9},
        {"content": "def auth():", "score": 0.9},  # exact duplicate
        {"content": "def login():", "score": 0.8},
    ]
    
    # Act
    deduped = deduplicate(chunks)
    
    # Assert
    assert len(deduped) == 2
    assert deduped[0]["content"] == "def auth():"
```

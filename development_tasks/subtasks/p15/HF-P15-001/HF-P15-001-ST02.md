## Тест-дизайн

```python
# test_file.py
class MyClass:
    """Docstring for class."""
    def method(self, x: int) -> str:
        """Method docstring."""
        return str(x)

def standalone_function(a, b):
    """Function docstring."""
    return a + b
```

```python
def test_extract_symbols():
    # Arrange
    extractor = SymbolExtractor()
    
    # Act
    symbols = extractor.extract_symbols(Path("test_file.py"))
    
    # Assert
    assert len(symbols) == 3
    class_sym = next(s for s in symbols if s.name == "MyClass")
    assert class_sym.type == "class"
    assert class_sym.docstring == "Docstring for class."
    
    func_sym = next(s for s in symbols if s.name == "standalone_function")
    assert func_sym.type == "function"
    assert func_sym.parameters == ["a", "b"]
```

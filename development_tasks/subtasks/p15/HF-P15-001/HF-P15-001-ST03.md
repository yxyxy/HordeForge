## Тест-дизайн

```python
def test_hybrid_retriever_merge():
    # Arrange
    retriever = HybridRetriever(vector_store=mock_vector, keyword_index=mock_keyword)
    
    vector_results = [
        {"id": "1", "score": 0.9, "content": "auth validation"},
        {"id": "2", "score": 0.8, "content": "login check"},
    ]
    keyword_results = [
        {"id": "2", "score": 1.0, "content": "login check"},
        {"id": "3", "score": 0.7, "content": "validate token"},
    ]
    
    # Act
    merged = retriever._merge_results(vector_results, keyword_results, alpha=0.5)
    
    # Assert
    assert merged[0]["id"] == "2"  # highest combined score
    assert len(merged) <= 4
```

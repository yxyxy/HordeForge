## Тест-дизайн

```python
def test_qdrant_connection():
    # Arrange
    store = QdrantStore(host="localhost", port=6333)
    
    # Act & Assert
    assert store.get_client() is not None
    assert store.collection_exists("repo_chunks")

def test_create_collection():
    # Arrange
    store = QdrantStore(host="localhost", port=6333)
    
    # Act
    store.create_collection("repo_chunks", vector_size=384)
    
    # Assert
    collection = store.get_collection("repo_chunks")
    assert collection.vectors_config.size == 384
```
